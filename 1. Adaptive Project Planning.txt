if (typeof window !== 'undefined') {
  const host = window.location.hostname.toLowerCase();

  const DEST: Record<string, string> = {
    'uat.commercialbanking.citidirect.com':
      'https://uat.commercialbanking.citidirect.com/ccb/eppublic/home/',
    'uat.citigroup.com':
      'https://uat.citigroup.com/global/businesses/banking/commercial-bank',
    // add prod if needed:
    'commercialbanking.citidirect.com':
      'https://commercialbanking.citidirect.com/ccb/eppublic/home/',
    'www.citigroup.com':
      'https://www.citigroup.com/global/businesses/banking/commercial-bank',
  };

  const url = DEST[host] ?? DEST['www.citigroup.com']; // fallback
  const tgt = (window.top && window.top !== window.self) ? window.top : window;

  try { (tgt as Window).location.href = url; }
  catch { window.location.href = url; }
}
