import React, { useState, useRef, useEffect } from 'react';

interface Option {
  label: string;
  value: string;
}

interface SelectProps {
  id: string;
  label: string;
  options: Option[];
  selectedValues?: string[];
  multiSelect?: boolean;
  searchable?: boolean;
  placeholder?: string;
  onChange?: (values: string[]) => void;
}

const CustomSelect: React.FC<SelectProps> = ({
  id,
  label,
  options,
  selectedValues = [],
  multiSelect = false,
  searchable = false,
  placeholder = 'Select...',
  onChange,
}) => {
  const [isOpen, setIsOpen] = useState(false);
  const [searchTerm, setSearchTerm] = useState('');
  const [selected, setSelected] = useState<string[]>(selectedValues);

  const containerRef = useRef<HTMLDivElement>(null);

  const toggleDropdown = () => setIsOpen((prev) => !prev);

  const handleSelect = (value: string) => {
    let newSelected;

    if (multiSelect) {
      if (selected.includes(value)) {
        newSelected = selected.filter((v) => v !== value);
      } else {
        newSelected = [...selected, value];
      }
    } else {
      newSelected = [value];
      setIsOpen(false);
    }

    setSelected(newSelected);
    onChange && onChange(newSelected);
  };

  const handleClickOutside = (event: MouseEvent) => {
    if (containerRef.current && !containerRef.current.contains(event.target as Node)) {
      setIsOpen(false);
    }
  };

  useEffect(() => {
    document.addEventListener('mousedown', handleClickOutside);
    return () => document.removeEventListener('mousedown', handleClickOutside);
  }, []);

  const filteredOptions = searchable
    ? options.filter((opt) => opt.label.toLowerCase().includes(searchTerm.toLowerCase()))
    : options;

  return (
    <div className="select-container" ref={containerRef}>
      <label id={`${id}-label`} htmlFor={id}>{label}</label>
      <div
        role="combobox"
        aria-haspopup="listbox"
        aria-expanded={isOpen}
        aria-controls={`${id}-listbox`}
        tabIndex={0}
        className="select-input"
        onClick={toggleDropdown}
        onKeyDown={(e) => { if (e.key === 'Enter') toggleDropdown(); }}
      >
        {multiSelect
          ? selected.length ? `${label} (${selected.length})` : placeholder
          : options.find((o) => o.value === selected[0])?.label || placeholder}
      </div>

      {isOpen && (
        <div className="dropdown-list" role="listbox" id={`${id}-listbox`}>
          {searchable && (
            <input
              type="text"
              placeholder="Search"
              value={searchTerm}
              onChange={(e) => setSearchTerm(e.target.value)}
              className="search-input"
            />
          )}

          <ul>
            {filteredOptions.map((option) => (
              <li
                key={option.value}
                role="option"
                aria-selected={selected.includes(option.value)}
                tabIndex={0}
                className={`dropdown-option ${selected.includes(option.value) ? 'selected' : ''}`}
                onClick={() => handleSelect(option.value)}
                onKeyDown={(e) => {
                  if (e.key === 'Enter' || e.key === ' ') handleSelect(option.value);
                }}
              >
                {multiSelect && (
                  <input type="checkbox" readOnly checked={selected.includes(option.value)} />
                )}
                {option.label}
              </li>
            ))}
          </ul>

          {multiSelect && (
            <button className="apply-btn" onClick={() => setIsOpen(false)}>Apply</button>
          )}
        </div>
      )}
    </div>
  );
};

export default CustomSelect;
