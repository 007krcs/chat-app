const buildDir = path.join(appDirectory, 'build');

app.use(['/global/CCBContactus/eppublic/contactus', '/ccb/eppublic/home'],
  express.static(buildDir));

app.get([
  '/global/CCBContactus/eppublic/contactus/*',
  '/ccb/eppublic/home/*'
], (_, res) => res.sendFile(path.join(buildDir, 'index.html')));

// (optional) if you want special cache headers for assets:
app.use(
  ['/global/CCBContactus/eppublic/contactus/static', '/ccb/eppublic/home/static'],
  express.static(path.join(buildDir, 'static'), {
    setHeaders: (res) => {
      res.setHeader('Cache-Control', 'no-store, no-cache');
      res.setHeader('Pragma', 'no-cache');
      res.setHeader('Expires', '0');
    },
  })
);
