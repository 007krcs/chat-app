const urlRegex = /^https?:\/\/[a-zA-Z0-9-]+(\.[a-zA-Z0-9-]+)+(\.[a-zA-Z]{2,})(\/[^\s]*)?$/;

// Example usage:
const validateUrl = (url) => urlRegex.test(url);

// Test cases
console.log(validateUrl("www.google"));          // false
console.log(validateUrl("abc.e"));               // false
console.log(validateUrl("test.n"));              // false
console.log(validateUrl("https://www.google.com")); // true
console.log(validateUrl("http://example.co.in")); // true
console.log(validateUrl("https://sub.example.com")); // true
console.log(validateUrl("http://localhost"));     // false
console.log(validateUrl("http://www.google"));    // false
console.log(validateUrl("https://google.com"));   // true
console.log(validateUrl("google"));               // false
console.log(validateUrl("http://google"));
